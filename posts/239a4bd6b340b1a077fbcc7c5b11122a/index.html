Python.sublime-build

{
    "cmd": ["python", "-u", "${file_name}"],
    "file_regex": "^[ ]*File \"(...*?)\", line ([0-9]*)",
    "working_dir": "${file_path}",
    "selector": "source.python",
    "env": {"PYTHONIOENCODING": "utf-8"},
    "variants":
    [
        {
            "name": "Python (cmd)",
            "shell_cmd": "start cmd /c \"python -u \"${file_name}\" & pause\""
        }
    ]
}
