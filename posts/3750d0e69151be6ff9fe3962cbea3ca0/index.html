int upper_bound(vector<int>& nums, int x) {
	int left = 0;
	int right = nums.size() - 1;

	while (left <= right) {
		int mid = left +(right - left) / 2;
		if (x >= nums[mid]) {       //è¿™é‡Œæ˜¯å¤§äºç­‰äº
			left = mid + 1;
		}
		else {
			right = mid - 1;	
		}
	}
	return left;
}
