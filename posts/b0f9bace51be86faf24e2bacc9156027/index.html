2.2、在QT中创建工程,配置环境

  1. 在工程目录下，创建bin文件夹，存放sqlite3.dll

  2. 在工程文件夹下创建lib文件，存放sqlite3.lib

  3. 在项目根目录下，放sqllite3.h

  4. 在pro文件末尾添加并保存：

DESTDIR=bin
LIBS+=$$PWD/lib/sqlite3.lib

  1. 将创建好的数据库文件拷贝到工程目录下的bin文件夹中。

3.3、代码测试

#include <QApplication>
#include "sqlite3.h"
#include <QDebug>
#include <QString>
int main(int argc, char *argv[])
{
    QApplication a(argc, argv);
    sqlite3 * dbconnect=nullptr;
    //1、打开数据库
    int res=sqlite3_open("./xxxxx.db",&dbconnect);
    if(res ==SQLITE_OK)
    {
        qDebug()<<"数据打开成功！";
    }else
    {
        qDebug()<<"数据库打开失败！";
        qDebug()<<sqlite3_errmsg(dbconnect);//打印错误信息
        qDebug()<<sqlite3_errcode(dbconnect);//打印错误的代码
    }
     //2、关闭数据库
    sqlite3_close(dbconnect);

    return a.exec();
}
