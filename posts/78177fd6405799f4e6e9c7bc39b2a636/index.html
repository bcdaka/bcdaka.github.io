typescript 断言

/*  断言
确定后期一定是某种格式
可用于表单大对象初始化是空，赋值时有具体字段。前期断言是会是某种格式
*/



interface obj {
  name: string
  location: string
  age?: number
}

// 会报错
// let data: obj = {
// 
// } 
let data: obj = {

} as obj; // 断言data会有obj里的字段
data.name = "全易";
data.location = "北京";
data.age = 18



interface fnT {
  (n: number): string
  title: string
  age?: number
}
let fn: fnT = function (n: number) {
  return "123"
} as fnT; // 断言后面会添加title
fn.title = "哈哈哈"
