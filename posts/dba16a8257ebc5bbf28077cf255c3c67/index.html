class Solution {
    public int trailingZeroes(int n) {
        int ans = 0;
        //本质：每5个数有一个5的倍数，每25个数有一个25的倍数……
        int num = 5;
        while(n / num != 0) {
            ans += n/num;
            num *= 5;
        }
        return ans;
    }
}
