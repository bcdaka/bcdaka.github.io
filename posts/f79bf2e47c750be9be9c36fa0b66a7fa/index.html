封装的话采用的是插槽

    <div v-if="loading"  >
     <slot name="template">
        <div v-if="type === 'loading'">
            <VanLoading></VanLoading>
        </div>
        <div v-if="type === 'skeleton'">
            <VanSkeleton :row="10" ></VanSkeleton>
            <VanSkeleton avatar :row="5"></VanSkeleton>

        </div>
     </slot>   
  
    </div>
