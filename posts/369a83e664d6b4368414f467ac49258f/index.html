下面是计算offset的方法，用来最大点位置整个x轴的相对位置。

      function calculatePercentageOfMax(array) {
        // 找到数组中的最大值及其索引位置
        const max = Math.max(...array)
        const index = array.indexOf(max)

        // 计算最大值位置相对于数组长度的百分比
        const percentage = index / (array.length - 1)

        return percentage
      }
