class Solution {
public:
	vector<int> maxSlidingWindow(vector<int>& nums, int k) {
		int i = 0;
		std::deque<int> queMax;
		vector<int> vRet;
		for ( i = 0; i < k; i++)
		{
			while (queMax.size() && (nums[queMax.back()] <= nums[i]))
			{
				queMax.pop_back();
			}
			queMax.emplace_back(i);			
		}
		vRet.emplace_back(nums[queMax.front()]);
		for (; i < nums.size(); i++)
		{
			if (i - k == queMax.front())
			{
				queMax.pop_front();
			}
			while (queMax.size() && (nums[queMax.back()] <= nums[i]))
			{
				queMax.pop_back();
			}
			queMax.emplace_back(i);
			vRet.emplace_back(nums[queMax.front()]);
		}
		return vRet;
	}
};
