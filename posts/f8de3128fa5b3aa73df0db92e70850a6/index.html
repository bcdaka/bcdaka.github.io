Vue + Element项目中select下拉框label想要显示多个值

<el-select
        v-model="form.plantId"
        collapse-tags
        filterable
        placeholder="请选择品种种类"
        style="width: 270px;">
  <el-option
          v-for="item in plantIdArray"
          :key="item.id"
          :label="`${item.batchId}-${item.varietyName}`"
          :value="item.plantId">
  </el-option>
</el-select>

:label="${item.batchId}-${item.varietyName}"
