回溯算法（c++）

#include <iostream>
#include <iomanip>
using namespace std;
char b[110][110];
int n;
int cntt = 0;
int x[110],y[110];
int di[] = {0,1,0,-1};
int dj[] = {1,0,-1,0};
int cnt = 0;
void aaa(int,int);
int main()
{
	cin>>n;
	for(int i = 0;i<n;i++)
	{
		for(int j = 0;j<n;j++)
		{
			b[i][j] = '.';
		}
	}
	
	aaa(0,0);
	return 0;
}
void aaa(int i,int j)
{
	b[i][j] = '#';
	x[cnt] = i;
	y[cnt] = j;
	cnt++;
	
	if(i==n-1&&j==n-1)
	{
		cntt++;
		cout<<cntt<<":";
		for(int ii = 0;ii<cnt-1;ii++)
		{
			cout<<x[ii]+1<<","<<y[ii]+1<<" --> ";
		}
		cout<<x[cnt-1]+1<<","<<y[cnt-1]+1<<endl;
		x[cnt-1] = 0;
		y[cnt-1] = 0;
		
		return;
	}
	for(int qqq = 0;qqq<4;qqq++)
	{
		int ti = i+di[qqq];
		int tj = j+dj[qqq];
		if(ti>=0&&ti<n&&tj>=0&&tj<n&&b[ti][tj]=='.')
		{
			b[ti][tj] = '#';
			aaa(ti,tj);
			cnt--;
			b[ti][tj] = '.';
		}
	}
	
	return;
}

#include <iostream>
#include <iomanip>
using namespace std;
char b[110][110];
int n,m;
bool f = false;
int ma = -99999;

int di[] = {0,1,0,-1};
int dj[] = {1,0,-1,0};
int c[100];
void aaa(int,int,int);
int main()
{
	cin>>n>>m;
	for(int i = 0;i<n;i++)
	{
		for(int j = 0;j<m;j++)
		{
			cin>>b[i][j];
		}
	}
	
	aaa(0,0,1);
	cout<<ma;
	return 0;
}
void aaa(int i,int j,int cnt)
{
	c[(int)b[i][j]] = 1;
	ma = max(ma,cnt);
	for(int qqq = 0;qqq<4;qqq++)
	{
		int ti = i+di[qqq];
		int tj = j+dj[qqq];
		if(ti>=0&&ti<n&&tj>=0&&tj<m&&c[(int)b[ti][tj]]==0)
		{
			c[(int)b[ti][tj]] = 1;
			aaa(ti,tj,cnt+1);
			
			c[(int)b[ti][tj]] = 0;
		}
	}
	
	return;
}
