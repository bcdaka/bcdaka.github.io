Pair get(int v, vector<int>& a, vector<int>& b, int n) {
    ll sum = 0;
    ll cnt = 0;
    for (int i = 0; i < n; ++i) {
        if (a[i] >= v) {
            ll t = (a[i] - v) / b[i] + 1;
            cnt += t;
            sum += a[i] * t - t * (t - 1) / 2 * b[i];
        }
    }
    return Pair(cnt, sum);
}

    int n, k;
    cin >> n >> k;
    vector<int> a(n), b(n);

    while (l < r) {
        int x = (l + r) / 2;

        if (get(x, a, b, n).first <= k)  r = x;
        else l = x + 1;
    }

    Pair res = get(l, a, b, n);
    if (l > 0)  res.second += (k - res.first) * (l - 1);

}
