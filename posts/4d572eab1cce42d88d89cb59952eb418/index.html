process.env.VUE_APP_BASE_API]: { target: `https://vue.ruoyi.vip`,}

2.pathRewrite单引号中加入 /prod-api

pathRewrite: {

'^' + process.env.VUE_APP_BASE_API]: '/prod-api'

3.在.env.development中变更

 VUE_APP_BASE_API = '/prod-api'
