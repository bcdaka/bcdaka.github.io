docker部署rabbitMQ

如果用目录挂载会启动失败，要用数据卷挂载。

docker pull rabbitmq:3.8-management
#挂载数据卷 -v mq-plugins:/plugins \	
#设置主机名 --hostname mq \
docker run \
 -e RABBITMQ_DEFAULT_USER=rabbitmq \
 -e RABBITMQ_DEFAULT_PASS=1234 \
 -v mq-plugins:/plugins \
 --name mq \
 -p 15672:15672 \
 -p 5672:5672 \
 -d \
 rabbitmq:3.8-management
 
# 查看挂载的数据卷所在信息（目录）
docker inspect mq-plugins
