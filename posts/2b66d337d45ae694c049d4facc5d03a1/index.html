	bool IsBalance()
	{
		if (_root == nullptr)

		//æ£€æŸ¥æ ¹èŠ‚ç‚¹
		if (_root->_col == RED)

	bool Check(Node* root, int blackNum, const int refNum)

	void RotateR(Node* parent)
	{
		Node* subL = parent->_left;
		Node* subLR = subL->_right;

		if (subLR)
			subLR->_parent = parent;
		parent->_left = subLR;

		subL->_right = parent;

		if (parent == _root)

		else
		{
			//parentä¸ºä¸€é¢—å­æ ‘
			if (ppNode->_left == parent)
				ppNode->_left = subL;

				ppNode->_right = subL;

			subL->_parent = ppNode;
		}
	}

				ppNode->_left = subR;
			else
				ppNode->_right = subR;

			subR->_parent = ppNode;
		}
