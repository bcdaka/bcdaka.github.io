信奥C++，输入一个n*m的矩阵，在输入一个变量s，如果s==0,则水平翻转矩阵，否则垂直翻转矩阵

属于信奥题目，具体解法如下。已经通过调试，欢迎使用。

有更有解法欢迎在评论区沟通，谢谢！

#include <iostream>

using namespace std;

int main(){

    int m,n;

    cin>>m>>n;

    int matrix[100][100];

    for(int i=0;i<m;i++){

        for(int j=0;j<n;j++){

            cin>>matrix[i][j];

        }

    }

    int s;

    cin>>s;

    if(s==0){

       for(int i=m-1;i>=0;i--){

           for(int j=0;j<n;j++){

               cout<<matrix[i][j]<<" ";

           }

           cout<<endl;

       }

    }else{

        for(int i=0;i<m;i++){

            for(int j=n-1;j>=0;j--){

                cout<<matrix[i][j]<<" ";

            }

            cout<<endl;

        }

    }

    return 0;

}
