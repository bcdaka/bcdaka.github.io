	assert(pq);
	pq->phead = pq->ptail = NULL;

{
	assert(pq);

	QNode* pur = pq->phead;

	while (pur)
	{
		QNode* cur = pur->next;
		free(pur);
		pur = cur;

	pq->phead = pq->ptail = NULL;
	pq->size = 0;
}
